    // Digital Rain Effect
    const canvas = document.getElementById('digitalRain');
    const ctx = canvas.getContext('2d');

    function resizeCanvas() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
    }
    resizeCanvas();
    window.addEventListener('resize', resizeCanvas);

    const chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789$#@%&*";
    const fontSize = 14;
    const columns = Math.floor(canvas.width / fontSize);
    const drops = Array(columns).fill(1);

    function draw() {
        ctx.fillStyle = 'rgba(10, 10, 18, 0.05)';
        ctx.fillRect(0, 0, canvas.width, canvas.height);

        ctx.fillStyle = '#ff2d78';
        ctx.font = fontSize + 'px Space Grotesk';

        for (let i = 0; i < drops.length; i++) {
            const text = chars.charAt(Math.floor(Math.random() * chars.length));
            ctx.fillText(text, i * fontSize, drops[i] * fontSize);

            if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
                drops[i] = 0;
            }
            drops[i]++;
        }
    }

    if (!window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
        setInterval(draw, 50);
    }

    // Terminal Typing Effect for CLI
    const cliTyping = document.getElementById('cli-typing');
    const cliResponse = document.getElementById('cli-response');
    const command = "apex-agent init --protocol alpha-9";
    const responseText = "> Spawning node... OK";
    let index = 0;

    function typeCLI() {
        if (index < command.length) {
            cliTyping.textContent += command.charAt(index);
            index++;
            setTimeout(typeCLI, 100);
        } else {
            setTimeout(() => {
                cliResponse.textContent = responseText;
            }, 500);
        }
    }

    // Secondary Text Terminal Effect
    function typeEffect(element, text, speed = 30) {
        element.textContent = '';
        let i = 0;
        const timer = setInterval(() => {
            if (i < text.length) {
                element.textContent += text.charAt(i);
                i++;
            } else {
                clearInterval(timer);
            }
        }, speed);
    }

    // Initialize effects on load
    window.addEventListener('load', () => {
        if (!window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
            typeCLI();
            const heroPara = document.querySelector('.typing-effect');
            typeEffect(heroPara, heroPara.getAttribute('data-text'));
        }
    });

    // Hover tilt effect for cards
    document.querySelectorAll('.group').forEach(card => {
        card.addEventListener('mousemove', (e) => {
            if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return;
            const rect = card.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            
            const centerX = rect.width / 2;
            const centerY = rect.height / 2;
            
            const rotateX = (y - centerY) / 30;
            const rotateY = (centerX - x) / 30;
            
            card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
        });
        
        card.addEventListener('mouseleave', () => {
            card.style.transform = 'perspective(1000px) rotateX(0deg) rotateY(0deg)';
        });
    });

<script>
    // Liquid Background & Card Shine Tracking
    document.addEventListener('mousemove', (e) => {
        if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return;

        const x = e.clientX;
        const y = e.clientY;
        const width = window.innerWidth;
        const height = window.innerHeight;

        // Background Liquid Effect
        const bgX = (x / width) * 100;
        const bgY = (y / height) * 100;
        document.documentElement.style.setProperty('--bg-x', `${bgX}%`);
        document.documentElement.style.setProperty('--bg-y', `${bgY}%`);

        // Card Shine Effect
        const cards = document.querySelectorAll('.glass-card');
        cards.forEach(card => {
            const rect = card.getBoundingClientRect();
            const cardX = x - rect.left;
            const cardY = y - rect.top;
            
            card.style.setProperty('--mouse-x', `${cardX}px`);
            card.style.setProperty('--mouse-y', `${cardY}px`);

            // Interactive radial highlight for old style (fallback/extra glow)
            if (cardX > 0 && cardX < rect.width && cardY > 0 && cardY < rect.height) {
                card.style.background = `radial-gradient(circle at ${cardX}px ${cardY}px, rgba(255, 45, 120, 0.1) 0%, rgba(30, 30, 48, 0.4) 60%)`;
            } else {
                card.style.background = 'rgba(30, 30, 48, 0.4)';
            }
        });
    });

    // Handle smooth transition on theme change
    const themeBtn = document.querySelector('[onclick*="toggle(\'dark\')"]');
    if (themeBtn) {
        themeBtn.addEventListener('click', () => {
            document.body.style.transition = 'background-color 0.5s ease, color 0.5s ease';
        });
    }
</script>

        // Matrix-like Digital Rain Effect
        const canvas = document.getElementById('digitalRain');
        const ctx = canvas.getContext('2d');

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        resizeCanvas();
        window.addEventListener('resize', resizeCanvas);

        const chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789$#@%&*";
        const fontSize = 14;
        const columns = canvas.width / fontSize;
        const drops = Array(Math.floor(columns)).fill(1);

        function draw() {
            ctx.fillStyle = 'rgba(10, 10, 18, 0.05)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            ctx.fillStyle = '#ff2d78';
            ctx.font = fontSize + 'px Space Grotesk';

            for (let i = 0; i < drops.length; i++) {
                const text = chars.charAt(Math.floor(Math.random() * chars.length));
                ctx.fillText(text, i * fontSize, drops[i] * fontSize);

                if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
                    drops[i] = 0;
                }
                drops[i]++;
            }
        }

        setInterval(draw, 50);

        // Hover tilt effect for cards
        document.querySelectorAll('.group').forEach(card => {
            card.addEventListener('mousemove', (e) => {
                const rect = card.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                
                const centerX = rect.width / 2;
                const centerY = rect.height / 2;
                
                const rotateX = (y - centerY) / 20;
                const rotateY = (centerX - x) / 20;
                
                card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
            });
            
            card.addEventListener('mouseleave', () => {
                card.style.transform = 'perspective(1000px) rotateX(0deg) rotateY(0deg)';
            });
        });
    </script>
